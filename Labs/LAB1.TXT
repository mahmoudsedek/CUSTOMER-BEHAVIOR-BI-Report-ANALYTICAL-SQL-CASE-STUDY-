SELECT CUSTOMER_ID, ORDER_ID, ORDER_DATE,
    LEAD (ORDER_DATE, 1) OVER (PARTITION BY CUSTOMER_ID ORDER BY ORDER_DATE ASC) AS NXT_ORDER_CUST
FROM OE.ORDERS;

SELECT CUSTOMER_ID, ORDER_ID, ORDER_DATE,
    LAG (ORDER_DATE, 1) OVER (PARTITION BY CUSTOMER_ID ORDER BY ORDER_DATE DESC) AS NXT_ORDER_CUST
FROM OE.ORDERS;

SELECT CUSTOMER_ID, ORDER_ID, ORDER_DATE,
    MAX(ORDER_DATE) OVER (PARTITION BY CUSTOMER_ID ORDER BY ORDER_DATE ROWS BETWEEN CURRENT ROW AND 1 FOLLOWING) NXT_ORDER_CUST
FROM OE.ORDERS;

SELECT CUSTOMER_ID, ORDER_ID, ORDER_DATE,
    MAX(ORDER_DATE) OVER (PARTITION BY CUSTOMER_ID ORDER BY ORDER_DATE DESC ROWS BETWEEN 1 PRECEDING AND CURRENT ROW) NXT_ORDER_CUST
FROM OE.ORDERS;
